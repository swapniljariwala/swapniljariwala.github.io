import os
import pandas as pd
import xgboost as xgb

EXPR = 'xgb500'
TRAIN_PREDICTION_FILE = os.path.join('predictions', EXPR + '_pred.csv') # prediction on training set
SUBMISSION_FILE = os.path.join('predictions', EXPR + '_submission.csv') # this will be subitted
MODEL_FILE = os.path.join('models', EXPR + '_model_deault.pkl') # file saved for model persistence

print('Reading the train test')
train = pd.read_pickle('train.pkl')
test = pd.read_pickle('test.pkl')
x_col = list(train.columns)
x_col.remove('id')
x_col.remove('loss')
print('Read complete, starting with fit')

train_xgb = xgb.DMatrix(train[x_col], train['loss'])

params = {"objective": "reg:linear", "booster":"gbtree", }
estimator = xgb.train(dtrain=train_xgb,params=params)

sub = estimator.predict(xgb.DMatrix(test[x_col]))
